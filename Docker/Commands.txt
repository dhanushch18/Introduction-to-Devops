# Verify Docker installation
docker --version

# Pull an image from Docker Hub
docker pull nginx

# List images
docker images

# Run a container
docker run -d -p 8080:80 nginx

# See running containers
docker ps

# Stop a container
docker stop <container_id>

# Remove a container
docker rm <container_id>

# Remove an image
docker rmi <image_id>
